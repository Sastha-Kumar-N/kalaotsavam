  <div class="main" id="main">
    <div class="column events">
      <h2>Events</h2>
      <p data-rule="Rules for Story Writing: Use original stories, 30 min time limit.">Story Writing</p>
      <p data-rule="Rules for Oil Painting: Use only water/oil colors. No tracing allowed.">Oil Painting</p>
      <p data-rule="Rules for Debate: Topics will be announced 1 hour before the event.">Debate</p>
      <p data-rule="Rules for Poetry: Theme will be provided on the spot.">Poetry Writing</p>
    </div>

    <div class="column rules">
      <h2>Rules</h2>
      <p id="rulesContent">Click an event to see the rules</p>
    </div>
  </div>

  <script>
    const eventItems = document.querySelectorAll('.events p');
    const rulesContent = document.getElementById('rulesContent');

    eventItems.forEach(item => {
      item.addEventListener('click', () => {
        const ruleText = item.getAttribute('data-rule');
        rulesContent.textContent = ruleText;

        eventItems.forEach(el => el.classList.remove('active'));
        item.classList.add('active');
      });
    });
  </script>
